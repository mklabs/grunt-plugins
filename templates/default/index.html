    return "awesome!!!";
  });

var fs = require("fs"),
  crypto = require("crypto");

var hash = crypto.createHash("md5");
hash.update(fs.readFileSync("./path/to/file.css"));
console.log(hash.digest("hex"));
// >> e207ea1eeb616799f29d679e1fdc2415
